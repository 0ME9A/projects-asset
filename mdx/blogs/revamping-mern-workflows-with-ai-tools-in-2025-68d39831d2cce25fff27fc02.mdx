### Basic Integration with Node.js
Use AI to scaffold a Node.js server:
- **Code**:
  ```javascript
  // Prompt: "Create an Express server with MongoDB connection"
  const express = require('express');
  const mongoose = require('mongoose');
  const app = express();

  mongoose.connect('mongodb://localhost/merndb', { useNewUrlParser: true, useUnifiedTopology: true })
    .then(() => console.log('MongoDB connected'))
    .catch(err => console.error('MongoDB connection error:', err));

  app.use(express.json());

  const userSchema = new mongoose.Schema({
    username: String,
    email: String
  });
  const User = mongoose.model('User', userSchema);

  app.get('/api/users', async (req, res) => {
    const users = await User.find();
    res.json(users);
  });

  app.listen(3000, () => console.log('Server running on port 3000'));
  ```
- **Explanation**: AI generates a functional Express server with MongoDB, leveraging your backend skills.

### React Frontend Setup
Let AI assist with a React component:
- **Code**:
  ```javascript
  // Prompt: "Create a React component to display a user list"
  import React, { useState, useEffect } from 'react';
  import axios from 'axios';

  const UserList = () => {
    const [users, setUsers] = useState([]);
    const [loading, setLoading] = useState(true);

    useEffect(() => {
      axios.get('http://localhost:3000/api/users')
        .then(response => {
          setUsers(response.data);
          setLoading(false);
        })
        .catch(error => console.error('Error fetching users:', error));
    }, []);

    if (loading) return <p>Loading...</p>;

    return (
      <div>
        <h2>Users</h2>
        <ul>
          {users.map(user => (
            <li key={user._id}>{user.username} ({user.email})</li>
          ))}
        </ul>
      </div>
    );
  };

  export default UserList;
  ```
- **Explanation**: AI crafts a React component fetching data, aligning with your frontend expertise.

## AI-Assisted Coding Across MERN

### MongoDB Schema Design
Generate a complex schema with AI:
- **Code**:
  ```javascript
  // Prompt: "Design a MongoDB schema for an e-commerce product with reviews"
  const mongoose = require('mongoose');

  const reviewSchema = new mongoose.Schema({
    userId: { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true },
    rating: { type: Number, min: 1, max: 5, required: true },
    comment: String,
    createdAt: { type: Date, default: Date.now }
  });

  const productSchema = new mongoose.Schema({
    name: { type: String, required: true },
    description: String,
    price: { type: Number, required: true },
    stock: { type: Number, default: 0 },
    reviews: [reviewSchema]
  });

  const Product = mongoose.model('Product', productSchema);
  module.exports = Product;
  ```
- **Explanation**: AI creates a nested schema with validation, enhancing your MongoDB skills.

### Express Route Optimization
Automate route creation:
- **Code**:
  ```javascript
  // Prompt: "Create an Express route to add a product with input validation"
  const express = require('express');
  const Product = require('./models/Product');
  const router = express.Router();

  router.post('/products', async (req, res) => {
    const { name, description, price, stock } = req.body;

    if (!name || !price || price < 0) {
      return res.status(400).json({ error: 'Invalid product data' });
    }

    const product = new Product({ name, description, price, stock });
    await product.save();
    res.status(201).json(product);
  });

  module.exports = router;
  ```
- **Explanation**: AI adds validation and error handling, streamlining your Express work.

## Real-World Example: AI-Powered E-Commerce App

Build an AI-assisted app:
- **Code** (Server):
  ```javascript
  // Prompt: "Create an Express route to recommend products"
  const router = express.Router();

  router.get('/api/recommendations', async (req, res) => {
    const userId = req.query.userId;
    const user = await User.findById(userId).populate('purchases');
    const purchasedCategories = user.purchases.map(p => p.category);
    const recommendations = await Product.find({
      category: { $in: purchasedCategories },
      _id: { $nin: user.purchases.map(p => p._id) }
    }).limit(5);
    res.json(recommendations);
  });
  ```
- **Code** (React):
  ```javascript
  // Prompt: "Create a React component for product recommendations"
  import React, { useState, useEffect } from 'react';
  import axios from 'axios';

  const Recommendations = ({ userId }) => {
    const [recommendations, setRecommendations] = useState([]);

    useEffect(() => {
      axios.get(`http://localhost:3000/api/recommendations?userId=${userId}`)
        .then(res => setRecommendations(res.data))
        .catch(err => console.error('Recommendation error:', err));
    }, [userId]);

    return (
      <div>
        <h2>Recommended for You</h2>
        <ul>
          {recommendations.map(product => (
            <li key={product._id}>{product.name} - ${product.price}</li>
          ))}
        </ul>
      </div>
    );
  };

  export default Recommendations;
  ```
- **Explanation**: AI designs a recommendation system, integrating MongoDB queries with a React UI.
